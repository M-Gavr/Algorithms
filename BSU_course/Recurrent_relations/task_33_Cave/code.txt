tuple<int, int, int, int, int, int, int, int> dfs(
    int v, int p, vector<map<int, int>>& inner, vector<map<int, int>>& outer, vector<int>& order,
    vector<pair<int, int>>& lr) {
    if (inner[v].size() == 1) {
        lr[v] = {v, v};
        return {order[v], order[v], 0, 0, 0, v, v, v};
    }
    vector<tuple<int, int, int, int, int, int, int, int>> res;
    vector<int> as;
    for (auto [a, w] : inner[v]) {
        if (a == p) {
            continue;
        }
        res.push_back(dfs(a, v, inner, outer, order, lr));
    }
    if (res[0] > res[1]) {
        swap(res[0], res[1]);
    }
    lr[v] = {get<7>(res[0]), get<7>(res[1])};
    return {
      get<0>(res[0]),
      get<1>(res[1]),
      get<4>(res[0]) + get<2>(res[1]) + outer[get<6>(res[0])][get<5>(res[1])] +
          inner[v][get<7>(res[1])],
      get<4>(res[1]) + get<3>(res[0]) + outer[get<6>(res[0])][get<5>(res[1])] +
          inner[v][get<7>(res[0])],
      get<2>(res[0]) + get<3>(res[1]) + inner[v][get<7>(res[0])] + inner[v][get<7>(res[1])],
      get<5>(res[0]),
      get<6>(res[1]),
      v};
    // {m,M,l-c,r-c,l-r,mi,Mi,ci}
    // {0,1,2  ,3  ,4  ,5 ,6 ,7}
    // l-c = l.l-l.r-r.l-r.c-c
    // r-c = r.r-r.l-l.r-l.c-c
    // l-r = l.l-l.c-c-r.c-r.r
}

void gen_ans(int v, int from, int to, vector<pair<int, int>>& lr) {
    if (lr[v].first == v) {
        cout << v + 1 << " ";
        return;
    }
    if (from == 0 && to == 1) {
        gen_ans(lr[v].first, 0, 2, lr);
        cout << v + 1 << " ";
        gen_ans(lr[v].second, 2, 1, lr);
    } else if (from == 0 && to == 2) {
        gen_ans(lr[v].first, 0, 1, lr);
        gen_ans(lr[v].second, 0, 2, lr);
        cout << v + 1 << " ";
    } else if (from == 2 && to == 1) {
        cout << v + 1 << " ";
        gen_ans(lr[v].first, 2, 1, lr);
        gen_ans(lr[v].second, 0, 1, lr);
    }
}

void Solve() {
    int k, n;
    cin >> n >> k;
    vector<map<int, int>> inner(n), outer(k);
    for (int i = 0; i < 3 * n / 2; i++) {
        int a, b, w;
        cin >> a >> b >> w;
        --a;
        --b;
        if (a < k && b < k) {
            outer[a][b] = w;
            outer[b][a] = w;
        } else {
            inner[a][b] = w;
            inner[b][a] = w;
        }
    }
    vector<int> order(k);
    int c = 0;
    int cur = 0, prev = -1;
    for (;;) {
        order[cur] = c++;
        for (const int a : outer[cur] | views::keys) {
            if (a != prev) {
                prev = cur;
                cur = a;
                break;
            }
        }
        if (cur == 0) {
            break;
        }
    }
    const int c0 = (inner[0] | views::keys).front();
    vector<pair<int, int>> ans(n);
    auto x = dfs(c0, 0, inner, outer, order, ans);
    // {m,M,l-c,r-c,l-r,mi,Mi,ci}
    // {0,1,2  ,3  ,4  ,5 ,6 ,7}
    int best = min(
        {outer[0][get<5>(x)] + outer[0][get<6>(x)] + get<4>(x),
         get<2>(x) + outer[0][get<5>(x)] + inner[0][c0],
         get<3>(x) + outer[0][get<6>(x)] + inner[0][c0]});
    if (best == outer[0][get<5>(x)] + outer[0][get<6>(x)] + get<4>(x)) {
        cout << "1 ";
        gen_ans(c0, 0, 1, ans);
    } else if (best == get<2>(x) + outer[0][get<5>(x)] + inner[0][c0]) {
        cout << "1 ";
        gen_ans(c0, 0, 2, ans);
    } else {
        cout << "1 ";
        gen_ans(c0, 2, 1, ans);
    }
}
